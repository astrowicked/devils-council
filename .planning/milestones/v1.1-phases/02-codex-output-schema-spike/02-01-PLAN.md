import sys, json, os, datetime
run_id, item_num, label, mode, prompt_file, out_file, dur_ms, codex_ver, schema_path, exit_code = sys.argv[1:11]

schema_valid = None
schema_errors = None

rec = {

    "corpus_item": f"{item_num}-{label}",
    "mode": mode,
    "request_prompt": prompt,
    "response_raw": raw,
    "response_parsed": parsed,
    "schema_valid": schema_valid,
    "schema_validation_errors": schema_errors,

    "timestamp_iso": datetime.datetime.utcnow().isoformat() + "Z",
    "codex_version": codex_ver,
    "codex_exit_code": int(exit_code),
}
print(json.dumps(rec, ensure_ascii=False))
PYEOF
